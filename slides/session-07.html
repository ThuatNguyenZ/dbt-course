🛠️ Using dbt_utils

Common Functions

# 1. Surrogate Key
select
    {{ dbt_utils.surrogate_key(['order_id', 'product_id']) }} as order_product_id
from {{ ref('stg_orders') }}

# 2. Date Spine (generate date range)
{{ dbt_utils.date_spine(
    datepart="day",
    start_date="cast('2024-01-01' as date)",
    end_date="cast('2024-12-31' as date)"
) }}

# 3. Pivot
{{ dbt_utils.pivot(
    column='status',
    values=['PENDING', 'PAID', 'SHIPPED'],
    cmp='sum',
    prefix='revenue_'
) }}

# 4. Not Empty String
where {{ dbt_utils.not_empty_string('customer_name') }}

⚡ Incremental Models

Full Refresh vs Incremental

ApproachProsCons
Full Refresh✓ Simple✗ Slow, expensive
Incremental✓ Fast, cost-effective✗ More complex

Incremental Model Pattern

# models/marts/fct_daily_sales.sql
{{
    config(
        materialized='incremental',
        unique_key='order_date',
        incremental_strategy='merge',
    )
}}

with orders as (
    select * from {{ ref('int_orders__enriched') }}
    {% if is_incremental() %}
        -- Only get new/updated data
        where order_at >= (select max(order_date) - interval '7 days' from {{ this }})
    {% endif %}
)

select
    date_trunc('day', order_at)::date as order_date,
    count(*) as order_count,
    sum(gross_amount) as revenue
from orders
group by 1

Incremental Strategies

  • append: Only new rows (logs, events)
  • delete+insert: Replace overlapping data
  • merge: Update existing + insert new (default for PG)
  • insert_overwrite: Partition overwrite (BigQuery)
