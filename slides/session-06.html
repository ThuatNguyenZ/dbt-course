🌾 Jinja Templating Basics

Jinja syntax trong DBT

# Jinja là templating language cho phép embed code trong SQL

# 1. Variables
{{ variable_name }}

# 2. Functions/Macros
{{ my_macro(arg1, arg2) }}

# 3. Control Structures
{% if condition %}
    SELECT * FROM table_a
{% else %}
    SELECT * FROM table_b
{% endif %}

# 4. For Loops
{% for model in ref('models') %}
    SELECT * FROM {{ model }}
    {% if not loop.last %} UNION ALL {% endif %}
{% endfor %}

# 5. Comments
{# This is a comment #}

DBT Built-in Functions

ref()

Reference to another model

{{ ref('stg_orders') }}

source()

Reference to a source table

{{ source('shopee', 'orders') }}

config()

Set model configuration

{{ config(materialized='table') }}

is_incremental()

Check if incremental model

{% if is_incremental() %}

🎯 Jinja Execution Flow

# Before compilation (what you write):
select
    order_id,
    {{ calculate_vat('total_amount', 0.10) }} as vat_amount
from {{ ref('stg_orders') }}

# After compilation (what runs):
select
    order_id,
    round(total_amount * 0.10, 0) as vat_amount
from analytics.staging.stg_orders

📝 Writing Macros - Examples

Example 1: Simple VAT Calculation

# macros/calculate_vat.sql

{% macro calculate_vat(amount, vat_rate=0.10) %}
    round({{ amount }}::numeric * {{ vat_rate }}, 0)
{% endmacro %}

# Usage in a model:
# models/marts/fct_sales.sql

select
    order_id,
    total_amount,
    {{ calculate_vat('total_amount', 0.10) }} as vat_amount,
    total_amount - {{ calculate_vat('total_amount', 0.10) }} as net_amount
from {{ ref('stg_orders') }}

Example 2: Date Truncation

# macros/truncate_date.sql

{% macro truncate_date(date_column, date_part='day') %}
    date_trunc('{{ date_part }}', {{ date_column }}::timestamp)::timestamp
{% endmacro %}

# Usage:
select
    {{ truncate_date('order_at', 'day') }} as order_date,
    {{ truncate_date('order_at', 'month') }} as order_month,
    {{ truncate_date('order_at', 'week') }} as order_week
from {{ ref('stg_orders') }}

Example 3: Conditional Logic

# macros/get_order_status.sql

{% macro get_order_status(status_code) %}
    case {{ status_code }}
        when 'SHOPEE_PENDING' then 'PENDING'
        when 'SHOPEE_PAID' then 'PAID'
        when 'SHOPEE_SHIPPED' then 'SHIPPED'
        when 'SHOPEE_DELIVERED' then 'DELIVERED'
        when 'SHOPEE_CANCELLED' then 'CANCELLED'
        else 'UNKNOWN'
    end
{% endmacro %}

# Usage:
select
    order_id,
    raw_status,
    {{ get_order_status('raw_status') }} as standardized_status
from {{ ref('stg_orders') }}

⚠️ Common Mistakes

  • Quên {{ }} khi gọi macro trong SQL
  • Quên {% %} cho control structures
  • Hardcode values thay vì dùng parameters
  • Không test macros với edge cases

🚀 Advanced Macro Patterns

Example 4: Generate Date Range

# macros/generate_date_range.sql

{% macro generate_date_range(start_date, end_date) %}
    {% set start = modules.datetime.datetime.strptime(start_date, '%Y-%m-%d') %}
    {% set end = modules.datetime.datetime.strptime(end_date, '%Y-%m-%d') %}
    {% set date_array = [] %}

    {% set current = start %}
    {% while current <= end %}
        {% do date_array.append(current.strftime('%Y-%m-%d')) %}
        {% set current = current + modules.datetime.timedelta(days=1) %}
    {% endwhile %}

    {{ return(date_array) }}
{% endmacro %}

# Usage in a model:
{% set dates = generate_date_range('2024-01-01', '2024-01-07') %}

with order_data as (
    select * from {{ ref('stg_orders') }}
)

{% for date in dates %}
    select
        '{{ date }}'::date as order_date,
        count(*) as order_count
    from order_data
    where date(order_at) = '{{ date }}'
    {% if not loop.last %}union all{% endif %}
{% endfor %}

Example 5: Cross-database Macros

# macros/calculate_ltv.sql (generic)

{% macro calculate_ltv(total_revenue, order_count) %}
    {{ return(adapter.dispatch('calculate_ltv', 'dbt_shopee_tiktok')(total_revenue, order_count)) }}
{% endmacro %}

# macros/postgres/calculate_ltv.sql

{% macro postgres__calculate_ltv(total_revenue, order_count) %}
    nullif({{ total_revenue }}::numeric, 0) / nullif({{ order_count }}, 0)
{% endmacro %}

# macros/bigquery/calculate_ltv.sql

{% macro bigquery__calculate_ltv(total_revenue, order_count) %}
    safe_divide({{ total_revenue }}, {{ order_count }})
{% endmacro %}

Bonus Challenge

# Tạo macro cho complex business logic
{% macro calculate_customer_lifetime_value(customer_id) %}
    (
        select
            sum(gross_amount - discount) as total_revenue,
            count(*) as total_orders,
            sum(gross_amount - discount) / count(*) as avg_order_value,
            max(order_at) - min(order_at) as customer_tenure_days
        from {{ ref('int_orders__enriched') }}
        where customer_id = '{{ customer_id }}'
    )
{% endmacro %}

# Sử dụng trong model
select
    customer_id,
    {{ calculate_customer_lifetime_value('customer_id') }} as ltv_metrics
from {{ ref('stg_customers') }}
